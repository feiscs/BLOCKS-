    2. Estructura del Proyecto

    Una estructura de carpetas bien organizada es crucial para proyectos grandes. Sigue el patrón de feature-based organization:

    📁 FPS-Vulkan-Game/ ├── project.godot # Configuración principal ├── FPS-Vulkan-Game.csproj # Proyecto .NET ├── 📁 .godot/ # Cache del editor (no versionar) ├── 📁 assets/ # Assets importados │ ├── 📁 models/ # GLTF/GLB/FBX │ ├── 📁 textures/ # PNG, WebP, EXR │ ├── 📁 sounds/ # OGG, WAV │ ├── 📁 shaders/ # GLSL .gdshader │ ├── 📁 fonts/ # TTF, OTF │ └── 📁 materials/ # .tres materiales ├── 📁 src/ # Código fuente C# │ ├── 📁 Core/ # Sistemas globales (Autoload) │ ├── 📁 Player/ # Control jugador │ ├── 📁 Weapons/ # Sistema armas │ ├── 📁 Multiplayer/ # Red y matchmaking │ ├── 📁 AI/ # Inteligencia artificial │ ├── 📁 UI/ # Interfaces y HUD │ ├── 📁 Monetization/ # DLC, Season Pass, Tienda │ └── 📁 Utils/ # Helpers y extensiones ├── 📁 scenes/ # Escenas .tscn │ ├── 📁 levels/ # Mapas del juego │ ├── 📁 ui/ # Escenas de UI │ └── 📁 props/ # Objetos reutilizables ├── 📁 dlc/ # Contenido descargable ├── 📁 docs/ # Documentación └── 📁 tests/ # Tests unitarios
    📁 Nota: La carpeta src contiene todo el código C#. Godot compila automáticamente todos los archivos .cs encontrados en el proyecto.

    3.1 GameManager.cs (Autoload/Singleton)

    El GameManager es el cerebro del juego. Gestiona el estado global, datos del jugador y progreso.

    src/Core/GameManager.cs 
    using Godot;
using System;
using System.Collections.Generic;

public partial class GameManager : Node
{
    public static GameManager Instance { get; private set; }
    
    [Signal] public delegate void PlayerSpawnedEventHandler(PlayerController player);
    [Signal] public delegate void MatchStartedEventHandler(string matchId);
    [Signal] public delegate void MatchEndedEventHandler(string winner);
    [Signal] public delegate void CurrencyChangedEventHandler(int amount);
    [Signal] public delegate void StateChangedEventHandler(GameState newState);
    
    // Player Data
    public PlayerData CurrentPlayer { get; set; } = new();
    public GameState State { get; private set; } = GameState.MainMenu;
    
    // Monetization
    public int PremiumCurrency { get; set; } = 1000;  // Valorant Points equivalent
    public HashSet<string> OwnedDLCs { get; set; } = new();
    public int BattlePassTier { get; set; } = 0;
    public bool HasPremiumPass { get; set; } = false;
    
    // Settings
    public float MouseSensitivity { get; set; } = 0.5f;
    public int TargetFPS { get; set; } = 144;
    public bool VSync { get; set; } = true;
    public bool FullScreen { get; set; } = true;
    
    public override void _Ready()
    {
        if (Instance != null)
        {
            QueueFree();
            return;
        }
        
        Instance = this;
        ProcessMode = ProcessModeEnum.Always;
        
        // Check Vulkan support
        var renderingDevice = RenderingServer.GetRenderingDevice();
        if (renderingDevice == null)
        {
            PushError("Vulkan rendering device not available! Falling back to OpenGL.");
            // Optionally switch to compatibility renderer
            return;
        }
        
        GD.Print($"[GameManager] GPU: {renderingDevice.GetDeviceName()}");
        GD.Print($"[GameManager] Driver: {renderingDevice.GetDriverVersion()}");
        
        LoadSettings();
        LoadPlayerData();
        
        // Prevent multiple instances
        GetTree().AutoAcceptQuit = false;
    }
    
    public void ChangeState(GameState newState)
    {
        var oldState = State;
        State = newState;
        EmitSignal(SignalName.StateChanged, (int)newState);
        GD.Print($"[GameManager] State changed: {oldState} -> {newState}");
    }
    
    public void AddCurrency(int amount)
    {
        PremiumCurrency += amount;
        EmitSignal(SignalName.CurrencyChanged, PremiumCurrency);
        SavePlayerData();
        GD.Print($"[GameManager] Added {amount} VP. Total: {PremiumCurrency}");
    }
    
    public bool SpendCurrency(int amount)
    {
        if (PremiumCurrency < amount) 
        {
            GD.Print("[GameManager] Insufficient funds!");
            return false;
        }
        PremiumCurrency -= amount;
        EmitSignal(SignalName.CurrencyChanged, PremiumCurrency);
        SavePlayerData();
        return true;
    }
    
    public void UnlockDLC(string dlcId)
    {
        if (OwnedDLCs.Add(dlcId))
        {
            GD.Print($"[GameManager] DLC unlocked: {dlcId}");
            SavePlayerData();
        }
    }
    
    public bool HasDLC(string dlcId) => OwnedDLCs.Contains(dlcId);
    
    private void LoadSettings()
    {
        var config = new ConfigFile();
        var err = config.Load("user://settings.cfg");
        if (err != Error.Ok) 
        {
            GD.Print("[GameManager] No settings file found, using defaults");
            return;
        }
        
        MouseSensitivity = (float)config.GetValue("input", "mouse_sensitivity", 0.5f);
        TargetFPS = (int)config.GetValue("video", "target_fps", 144);
        VSync = (bool)config.GetValue("video", "vsync", true);
        FullScreen = (bool)config.GetValue("video", "fullscreen", true);
        
        ApplySettings();
    }
    
    public void SaveSettings()
    {
        var config = new ConfigFile();
        config.SetValue("input", "mouse_sensitivity", MouseSensitivity);
        config.SetValue("video", "target_fps", TargetFPS);
        config.SetValue("video", "vsync", VSync);
        config.SetValue("video", "fullscreen", FullScreen);
        
        config.Save("user://settings.cfg");
        ApplySettings();
    }
    
    private void ApplySettings()
    {
        Engine.MaxFps = TargetFPS;
        DisplayServer.WindowSetVsyncMode(VSync ? DisplayServer.VSyncMode.Enabled : DisplayServer.VSyncMode.Disabled);
        
        if (FullScreen)
            DisplayServer.WindowSetMode(DisplayServer.WindowMode.Fullscreen);
        else
            DisplayServer.WindowSetMode(DisplayServer.WindowMode.Windowed);
    }
    
    private void LoadPlayerData()
    {
        if (!FileAccess.FileExists("user://player.save")) return;
        
        using var file = FileAccess.Open("user://player.save", FileAccess.ModeFlags.Read);
        var json = file.GetAsText();
        var data = Json.ParseString(json).AsGodotDictionary();
        
        PremiumCurrency = (int)data.GetValueOrDefault("vp", 1000);
        BattlePassTier = (int)data.GetValueOrDefault("bp_tier", 0);
        HasPremiumPass = (bool)data.GetValueOrDefault("bp_premium", false);
        CurrentPlayer.Username = (string)data.GetValueOrDefault("username", "Player");
        CurrentPlayer.UserId = (string)data.GetValueOrDefault("user_id", Guid.NewGuid().ToString());
        
        var dlcs = data.GetValueOrDefault("dlcs", new Godot.Collections.Array<string>());
        foreach (var dlc in dlcs)
            OwnedDLCs.Add(dlc.ToString());
            
        GD.Print($"[GameManager] Loaded player: {CurrentPlayer.Username}");
    }
    
    public void SavePlayerData()
    {
        var data = new Godot.Collections.Dictionary
        {
            ["vp"] = PremiumCurrency,
            ["bp_tier"] = BattlePassTier,
            ["bp_premium"] = HasPremiumPass,
            ["username"] = CurrentPlayer.Username,
            ["user_id"] = CurrentPlayer.UserId,
            ["dlcs"] = new Godot.Collections.Array<string>(OwnedDLCs)
        };
        
        using var file = FileAccess.Open("user://player.save", FileAccess.ModeFlags.Write);
        file.StoreString(Json.Stringify(data));
    }
    
    public override void _Notification(int what)
    {
        if (what == NotificationWMCloseRequest)
        {
            SavePlayerData();
            SaveSettings();
            GetTree().Quit();
        }
    }
}

    using Godot;
using System;

[GlobalClass]
public partial class PlayerController : CharacterBody3D
{
    [ExportGroup("Movement Settings")]
    [Export] public float WalkSpeed = 5.0f;
    [Export] public float SprintSpeed = 8.0f;
    [Export] public float CrouchSpeed = 2.5f;
    [Export] public float JumpVelocity = 4.5f;
    [Export] public float Acceleration = 10.0f;
    [Export] public float Friction = 10.0f;
    [Export] public float AirControl = 0.3f;
    [Export] public float MouseSensitivity = 0.005f;
    
    [ExportGroup("Camera")]
    [Export] public Camera3D Camera;
    [Export] public Node3D CameraPivot;
    [Export] public ShapeCast3D CrouchCheck;
    [Export] public float CameraHeightStanding = 1.6f;
    [Export] public float CameraHeightCrouching = 0.8f;
    
    [ExportGroup("Weapons")]
    [Export] public Node3D WeaponHolder;
    [Export] public RayCast3D AimRay;
    [Export] public float WeaponSwayAmount = 0.05f;
    
    // State
    public bool IsSprinting { get; private set; }
    public bool IsCrouching { get; private set; }
    public bool IsAiming { get; private set; }
    public bool IsGrounded => IsOnFloor();
    public bool IsMoving => Velocity.Length() > 0.1f;
    
    private float _gravity;
    private float _currentSpeed;
    private float _cameraVerticalRotation = 0f;
    private Vector3 _velocity;
    private Vector3 _weaponBasePosition;
    private Vector3 _weaponBaseRotation;
    
    // Components
    private WeaponManager _weaponManager;
    private HealthSystem _healthSystem;
    private AnimationPlayer _animPlayer;
    private CollisionShape3D _collisionShape;
    private CapsuleShape3D _capsuleShape;
    
    [Signal] public delegate void DiedEventHandler(Node3D killer);
    [Signal] public delegate void HealthChangedEventHandler(int current, int max);
    [Signal] public delegate void LandedEventHandler(float velocity);
    
    public override void _Ready()
    {
        _gravity = (float)ProjectSettings.GetSetting("physics/3d/default_gravity");
        _currentSpeed = WalkSpeed;
        
        // Get components
        _weaponManager = GetNode<WeaponManager>("WeaponManager");
        _healthSystem = GetNode<HealthSystem>("HealthSystem");
        _animPlayer = GetNode<AnimationPlayer>("AnimationPlayer");
        _collisionShape = GetNode<CollisionShape3D>("CollisionShape3D");
        _capsuleShape = _collisionShape.Shape as CapsuleShape3D;
        
        // Setup weapon
        if (WeaponHolder != null)
        {
            _weaponBasePosition = WeaponHolder.Position;
            _weaponBaseRotation = WeaponHolder.Rotation;
        }
        
        // Lock mouse
        Input.MouseMode = Input.MouseModeEnum.Captured;
        
        // Load sensitivity
        if (GameManager.Instance != null)
            MouseSensitivity = GameManager.Instance.MouseSensitivity;
            
        // Connect health events
        if (_healthSystem != null)
            _healthSystem.HealthChanged += OnHealthChanged;
    }
    
    public override void _Input(InputEvent @event)
    {
        if (@event is InputEventMouseMotion mouseMotion && Input.MouseMode == Input.MouseModeEnum.Captured)
        {
            HandleMouseLook(mouseMotion);
        }
        
        if (@event.IsActionPressed("ui_cancel"))
        {
            ToggleMouseCapture();
        }
    }
    
    private void HandleMouseLook(InputEventMouseMotion motion)
    {
        // Horizontal rotation (player body)
        RotateY(-motion.Relative.X * MouseSensitivity);
        
        // Vertical rotation (camera)
        _cameraVerticalRotation -= motion.Relative.Y * MouseSensitivity;
        _cameraVerticalRotation = Mathf.Clamp(_cameraVerticalRotation, -Mathf.Pi / 2.2f, Mathf.Pi / 2.2f);
        
        if (CameraPivot != null)
            CameraPivot.Rotation = new Vector3(_cameraVerticalRotation, 0, 0);
            
        // Weapon sway
        ApplyWeaponSway(motion.Relative);
    }
    
    private void ApplyWeaponSway(Vector2 mouseDelta)
    {
        if (WeaponHolder == null) return;
        
        var swayX = Mathf.Clamp(mouseDelta.X * WeaponSwayAmount, -0.1f, 0.1f);
        var swayY = Mathf.Clamp(mouseDelta.Y * WeaponSwayAmount, -0.1f, 0.1f);
        
        WeaponHolder.Rotation = _weaponBaseRotation + new Vector3(swayY, swayX, 0);
    }
    
    public override void _PhysicsProcess(double delta)
    {
        float dt = (float)delta;
        
        HandleMovement(dt);
        HandleActions();
        ApplyGravity(dt);
        UpdateCameraHeight(dt);
        
        Velocity = _velocity;
        MoveAndSlide();
        
        // Handle landing
        if (IsOnFloor() && _velocity.Y < -5f)
            EmitSignal(SignalName.Landed, -_velocity.Y);
        
        // Sync weapon
        _weaponManager?.UpdateWeaponPosition(IsAiming, dt);
    }
    
    private void HandleMovement(float delta)
    {
        Vector3 inputDir = Vector3.Zero;
        
        // Get input
        if (Input.IsActionPressed("move_forward")) inputDir -= Transform.Basis.Z;
        if (Input.IsActionPressed("move_backward")) inputDir += Transform.Basis.Z;
        if (Input.IsActionPressed("move_left")) inputDir -= Transform.Basis.X;
        if (Input.IsActionPressed("move_right")) inputDir += Transform.Basis.X;
        
        inputDir = inputDir.Normalized();
        
        // Sprint logic
        bool wantsSprint = Input.IsActionPressed("sprint") && !IsCrouching && inputDir != Vector3.Zero;
        if (wantsSprint != IsSprinting)
        {
            IsSprinting = wantsSprint;
            _currentSpeed = IsSprinting ? SprintSpeed : (IsCrouching ? CrouchSpeed : WalkSpeed);
        }
        
        // Crouch toggle
        if (Input.IsActionJustPressed("crouch"))
            ToggleCrouch();
        
        // Calculate target velocity
        Vector3 targetVelocity = inputDir * _currentSpeed;
        
        // Apply acceleration/friction
        if (IsOnFloor())
        {
            if (inputDir != Vector3.Zero)
                _velocity.X = Mathf.MoveToward(_velocity.X, targetVelocity.X, Acceleration * delta);
                _velocity.Z = Mathf.MoveToward(_velocity.Z, targetVelocity.Z, Acceleration * delta);
            else
                _velocity.X = Mathf.MoveToward(_velocity.X, 0, Friction * delta);
                _velocity.Z = Mathf.MoveToward(_velocity.Z, 0, Friction * delta);
        }
        else
        {
            // Air control
            _velocity.X = Mathf.MoveToward(_velocity.X, targetVelocity.X, Acceleration * AirControl * delta);
            _velocity.Z = Mathf.MoveToward(_velocity.Z, targetVelocity.Z, Acceleration * AirControl * delta);
        }
        
        // Jump
        if (Input.IsActionJustPressed("jump") && IsOnFloor() && !IsCrouching)
        {
            _velocity.Y = JumpVelocity;
            _animPlayer?.Play("jump");
        }

        if (inputDir != Vector3.Zero && IsOnFloor())

public abstract partial class BaseWeapon : Node3D
{

    [Export] public string WeaponName = "Weapon";
    [Export] public string WeaponId = "weapon_default";

    [Export] public int HeadshotDamage = 75;
    [Export] public int FireRate = 600;  // RPM (Rounds Per Minute)
    [Export] public int MagazineSize = 30;
    [Export] public int ReserveAmmo = 90;
    [Export] public float ReloadTime = 2.0f;
    [Export] public float Range = 100f;
    [Export] public WeaponType Type = WeaponType.Rifle;
    [Export] public FireMode FireMode = FireMode.Auto;
    
    [ExportGroup("Recoil Settings")]
    [Export] public float RecoilVertical = 0.5f;
    [Export] public float RecoilHorizontal = 0.2f;
    [Export] public float RecoilRecovery = 5f;
    [Export] public float AimRecoilMultiplier = 0.5f;
    
    [ExportGroup("Spread Settings")]
    [Export] public float BaseSpread = 0.01f;
    [Export] public float MoveSpread = 0.05f;
    [Export] public float JumpSpread = 0.1f;
    [Export] public float AimSpreadMultiplier = 0.3f;

    [Export] public AudioStream ReloadSound;
    [Export] public AudioStream EmptySound;
    [Export] public float SoundMaxDistance = 50f;
    
    // State
    public int CurrentAmmo { get; protected set; }
    public int CurrentReserve { get; protected set; }
    public bool IsReloading { get; protected set; }
    public bool CanShoot => !IsReloading && CurrentAmmo > 0 && _fireTimer <= 0 && !_isBursting;
    public float FireInterval => 60f / FireRate;
    
    protected float _fireTimer = 0f;
    protected float _recoilAccumulation = 0f;
    protected bool _isBursting = false;
    protected int _burstCount = 0;
    protected RandomNumberGenerator _rng = new();
    
    // Components
    protected AudioStreamPlayer3D _audioPlayer;
    protected AnimationPlayer _animPlayer;
    protected GpuParticles3D _muzzleFlash;
    protected RayCast3D _fireRay;
    protected Node3D _muzzlePoint;
    
    [Signal] public delegate void FiredEventHandler();
    [Signal] public delegate void ReloadedEventHandler();
    [Signal] public delegate void EmptyEventHandler();
    [Signal] public delegate void AmmoChangedEventHandler(int current, int reserve);
    
    public override void _Ready()
    {
        CurrentAmmo = MagazineSize;
        CurrentReserve = ReserveAmmo;
        _rng.Randomize();
        
        // Get components
        _audioPlayer = GetNodeOrNull<AudioStreamPlayer3D>("AudioStreamPlayer3D");
        _animPlayer = GetNodeOrNull<AnimationPlayer>("AnimationPlayer");
        _muzzleFlash = GetNodeOrNull<GpuParticles3D>("MuzzleFlash");
        _fireRay = GetNodeOrNull<RayCast3D>("FireRay");
        _muzzlePoint = GetNodeOrNull<Node3D>("MuzzlePoint");
        
        if (_audioPlayer == null)
        {
            _audioPlayer = new AudioStreamPlayer3D();
            _audioPlayer.Name = "AudioStreamPlayer3D";
            _audioPlayer.MaxDistance = SoundMaxDistance;
            AddChild(_audioPlayer);
        }
    }
    
    public override void _Process(double delta)
    {
        float dt = (float)delta;
        
        if (_fireTimer > 0)
            _fireTimer -= dt;
        
        // Recoil recovery
        _recoilAccumulation = Mathf.MoveToward(_recoilAccumulation, 0, RecoilRecovery * dt);
    }
    
    public virtual bool TryShoot(bool isAiming)
    {
        if (!CanShoot)
        {
            if (CurrentAmmo == 0 && _fireTimer <= 0)
            {
                PlayEmptySound();
                EmitSignal(SignalName.Empty);
            }
            return false;
        }
        
        // Handle fire modes
        switch (FireMode)
        {
            case FireMode.Single:
            case FireMode.Burst:
                if (!Input.IsActionJustPressed("shoot")) return false;
                break;
            case FireMode.Auto:
                if (!Input.IsActionPressed("shoot")) return false;
                break;
        }
        
        _fireTimer = FireInterval;
        
        // Burst handling
        if (FireMode == FireMode.Burst)
        {
            _isBursting = true;
            _burstCount = 3;
            FireBurstShot(isAiming);
        }
        else
        {
            FireSingleShot(isAiming);
        }
        
        return true;
    }
    
    protected virtual void FireSingleShot(bool isAiming)
    {
        // Consume ammo
        CurrentAmmo--;
        EmitSignal(SignalName.AmmoChanged, CurrentAmmo, CurrentReserve);
        
        // Effects
        PlayShootEffects();
        
        // Raycast
        PerformRaycast(isAiming);
        
        // Recoil
        ApplyRecoil(isAiming);
        
        EmitSignal(SignalName.Fired);
    }
    
    protected async void FireBurstShot(bool isAiming)
    {
        for (int i = 0; i < _burstCount; i++)
        {
            if (CurrentAmmo <= 0) break;
            
            FireSingleShot(isAiming);
            await ToSignal(GetTree().CreateTimer(FireInterval), "timeout");
        }
        _isBursting = false;
    }
    
    protected virtual void PerformRaycast(bool isAiming)
    {
        if (_fireRay == null) return;
        
        float spread = CalculateSpread(isAiming);
        
        // Apply spread
        _fireRay.Rotation = new Vector3(
            _rng.Randf() * spread - spread / 2,
            _rng.Randf() * spread - spread / 2,
            0
        );
        
        _fireRay.ForceRaycastUpdate();
        
        if (_fireRay.IsColliding())
        {
            var hit = _fireRay.GetCollider() as Node3D;
            var hitPoint = _fireRay.GetCollisionPoint();
            var normal = _fireRay.GetCollisionNormal();
            
            // Spawn impact
            SpawnImpactEffect(hitPoint, normal);
            
            // Damage
            if (hit is PlayerController player)
            {
                bool isHeadshot = CheckHeadshot(hitPoint, player);
                int damage = isHeadshot ? HeadshotDamage : Damage;
                
                player.TakeDamage(damage, hitPoint, GetOwner() as Node3D, isHeadshot);
                ShowHitMarker(isHeadshot);
                
                // Stats
                if (isHeadshot && GameManager.Instance != null)
                    GameManager.Instance.CurrentPlayer.Stats.Headshots++;
            }
            else if (hit is Damageable damageable)
            {
                damageable.TakeDamage(Damage, hitPoint);
            }
        }
    }
    
    protected float CalculateSpread(bool isAiming)
    {
        float spread = BaseSpread;
        
        // Movement spread
        var player = GetOwner() as PlayerController;
        if (player != null)
        {
            if (!player.IsGrounded) spread += JumpSpread;
            if (player.IsMoving) spread += MoveSpread;
        }
        
        // Aiming reduces spread
        if (isAiming) spread *= AimSpreadMultiplier;
        
        // Recoil adds spread
        spread += _recoilAccumulation * 0.05f;
        
        return spread;
    }
    
    protected void ApplyRecoil(bool isAiming)
    {
        float recoilMult = isAiming ? AimRecoilMultiplier : 1.0f;
        _recoilAccumulation += RecoilVertical * recoilMult;
        
        // Apply to camera
        var player = GetOwner() as PlayerController;
        if (player?.Camera != null)
        {
            float recoilX = RecoilVertical * recoilMult;
            float recoilY = _rng.Randf() * RecoilHorizontal * 2 - RecoilHorizontal;
            player.Camera.Rotation -= new Vector3(recoilX, recoilY, 0);
        }
    }
    
    public virtual void Reload()
    {
        if (IsReloading || CurrentAmmo >= MagazineSize || CurrentReserve <= 0)
            return;
        
        IsReloading = true;
        
        _animPlayer?.Play("reload");
        
        if (_audioPlayer != null && ReloadSound != null)
        {
            _audioPlayer.Stream = ReloadSound;
            _audioPlayer.Play();
        }
        
        GetTree().CreateTimer(ReloadTime).Timeout += FinishReload;
    }
    
    protected virtual void FinishReload()
    {
        int ammoNeeded = MagazineSize - CurrentAmmo;
        int ammoToLoad = Mathf.Min(ammoNeeded, CurrentReserve);
        
        CurrentAmmo += ammoToLoad;
        CurrentReserve -= ammoToLoad;
        
        IsReloading = false;
        EmitSignal(SignalName.Reloaded);
        EmitSignal(SignalName.AmmoChanged, CurrentAmmo, CurrentReserve);
    }
    
    protected void PlayShootEffects()
    {
        _animPlayer?.Play("shoot");
        
        if (_audioPlayer != null && ShootSound != null)
        {
            _audioPlayer.Stream = ShootSound;
            _audioPlayer.Play();
        }
        
        _muzzleFlash?.Restart();
        _muzzleFlash?.Emitting = true;
        
        // Shell ejection
        EjectShell();
    }
    
    protected void PlayEmptySound()
    {
        if (_audioPlayer != null && EmptySound != null)
        {
            _audioPlayer.Stream = EmptySound;
            _audioPlayer.Play();
        }
    }
    
    protected void SpawnImpactEffect(Vector3 position, Vector3 normal)
    {
        // Determine surface type and spawn appropriate effect
        var impact = GD.Load<PackedScene>("res://effects/bullet_impact.tscn");
        if (impact != null)
        {
            var instance = impact.Instantiate<Node3D>();
            instance.GlobalPosition = position;
            instance.LookAt(position + normal, Vector3.Up);
            GetTree().CurrentScene?.AddChild(instance);
        }
    }
    
    protected void EjectShell()
    {
        // Spawn shell casing
        var shell = GD.Load<PackedScene>("res://effects/shell_casing.tscn");
        if (shell != null && _muzzlePoint != null)
        {
            var instance = shell.Instantiate<RigidBody3D>();
            instance.GlobalPosition = _muzzlePoint.GlobalPosition;
            instance.ApplyImpulse(-_muzzlePoint.GlobalTransform.Basis.X * 2 + Vector3.Up * 0.5f);
            GetTree().CurrentScene?.AddChild(instance);
        }
    }
    
    protected void ShowHitMarker(bool isHeadshot)
    {
        var hud = GetNode<Control>("/root/PlayerHUD");
        hud?.Call("ShowHitMarker", isHeadshot);
    }
    
    protected bool CheckHeadshot(Vector3 hitPoint, PlayerController player)
    {
        var headPos = player.GlobalPosition + new Vector3(0, 1.6f, 0);
        return hitPoint.DistanceTo(headPos) < 0.15f;
    }

    {
        CurrentReserve += amount;
        EmitSignal(SignalName.AmmoChanged, CurrentAmmo, CurrentReserve);
    }

    using Godot;
using System;
using System.Collections.Generic;

public partial class NetworkManager : Node
{
    public static NetworkManager Instance { get; private set; }
    
    [Signal] public delegate void PlayerConnectedEventHandler(int peerId, string username);
    [Signal] public delegate void PlayerDisconnectedEventHandler(int peerId);
    [Signal] public delegate void ServerDisconnectedEventHandler();
    [Signal] public delegate void MatchReadyEventHandler();
    [Signal] public delegate void ConnectionFailedEventHandler(string error);
    
    public const int DEFAULT_PORT = 7777;
    public const int MAX_PLAYERS = 10;
    public const float NETWORK_TICK_RATE = 20.0f; // 20 ticks per second
    
    public bool IsServer => Multiplayer.IsServer();
    public bool IsConnected => Multiplayer.HasMultiplayerPeer();
    public int LocalPeerId => Multiplayer.GetUniqueId();
    public string ServerAddress { get; private set; } = "";
    
    private Dictionary<int, NetworkPlayerData> _players = new();
    private ENetMultiplayerPeer _peer;
    private double _networkTimer = 0;
    
    public override void _Ready()
    {
        if (Instance != null)
        {
            QueueFree();
            return;
        }
        
        Instance = this;
        ProcessMode = ProcessModeEnum.Always;
        
        // Connect multiplayer signals
        Multiplayer.ConnectedToServer += OnConnectedToServer;
        Multiplayer.ConnectionFailed += OnConnectionFailed;
        Multiplayer.PeerConnected += OnPeerConnected;
        Multiplayer.PeerDisconnected += OnPeerDisconnected;
        Multiplayer.ServerDisconnected += OnServerDisconnected;

    public Error CreateServer(int port = DEFAULT_PORT, int maxPlayers = MAX_PLAYERS)
    {
        _peer = new ENetMultiplayerPeer();
        var error = _peer.CreateServer(port, maxPlayers);
        
        if (error != Error.Ok)
        {
            GD.PushError($"[NetworkManager] Failed to create server: {error}");
            EmitSignal(SignalName.ConnectionFailed, error.ToString());
            return error;
        }
        
        _peer.Host.Compress(ENetConnection.CompressionMode.RangeCoder);
        Multiplayer.MultiplayerPeer = _peer;
        
        // Add host player (ID 1)
        RegisterPlayer(1, GameManager.Instance?.CurrentPlayer?.Username ?? "Host");
        
        GD.Print($"[NetworkManager] Server created on port {port}");
        return Error.Ok;
    }
    
    public Error JoinServer(string address, int port = DEFAULT_PORT)
    {
        ServerAddress = address;
        _peer = new ENetMultiplayerPeer();
        var error = _peer.CreateClient(address, port);
        
        if (error != Error.Ok)
        {
            GD.PushError($"[NetworkManager] Failed to create client: {error}");
            EmitSignal(SignalName.ConnectionFailed, error.ToString());
            return error;
        }
        
        _peer.Host.Compress(ENetConnection.CompressionMode.RangeCoder);
        Multiplayer.MultiplayerPeer = _peer;
        
        GD.Print($"[NetworkManager] Connecting to {address}:{port}...");
        return Error.Ok;
    }

    {
        Multiplayer.MultiplayerPeer = null;
        _peer?.Close();
        _peer = null;
        _players.Clear();
        GD.Print("[NetworkManager] Disconnected");
    }
    
    public override void _PhysicsProcess(double delta)
    {
        // Network tick for server reconciliation
        _networkTimer += delta;
        if (_networkTimer >= 1.0 / NETWORK_TICK_RATE)
        {
            _networkTimer = 0;
            OnNetworkTick();
        }
    }
    
    private void OnNetworkTick()
    {
        if (!IsConnected) return;
        
        // Server: broadcast game state
        if (IsServer)
        {
            BroadcastGameState();
        }
    }
    
    private void BroadcastGameState()
    {
        // Collect player states
        var playerStates = new Dictionary<int, PlayerState>();
        foreach (var kvp in _players)
        {
            var player = GetPlayerNode(kvp.Key);
            if (player != null)
            {
                playerStates[kvp.Key] = new PlayerState
                {
                    Position = player.GlobalPosition,
                    Rotation = player.Rotation,
                    Velocity = player.Velocity,
                    Health = player.GetHealth()
                };
            }
        }
        
        // Broadcast to all clients
        Rpc(nameof(SyncGameState), playerStates);
    }
    
    [Rpc(MultiplayerApi.RpcMode.Authority, CallLocal = false)]

    public void RequestShoot(int peerId, Vector3 origin, Vector3 direction)
    {
        if (!IsServer) return;
        
        var shooter = GetPlayerNode(peerId);
        if (shooter == null) return;
        
        // Validate origin (anti-cheat)
        if (origin.DistanceTo(shooter.GlobalPosition) > 5f)
        {
            GD.Print($"[AntiCheat] Player {peerId} position mismatch!");
            return;
        }
        
        // Perform raycast on server
        var spaceState = GetWorld3D().DirectSpaceState;
        var query = PhysicsRayQueryParameters3D.Create(origin, origin + direction * 1000);
        query.CollisionMask = 0b101; // Players and environment
        query.Exclude = new Godot.Collections.Array<Rid> { shooter.GetRid() };
        
        var result = spaceState.IntersectRay(query);
        
        if (result.Count > 0)
        {
            var hit = result["collider"].AsGodotObject() as Node3D;
            var hitPoint = result["position"].AsVector3();
            var hitNormal = result["normal"].AsVector3();
            
            if (hit is MultiplayerPlayer target && target.PeerId != peerId)
            {
                bool isHeadshot = IsHeadshot(hitPoint, target);
                int damage = isHeadshot ? 75 : 25;
                
                // Apply damage
                target.Rpc(nameof(target.TakeDamage), damage, peerId, isHeadshot);
                
                // Confirm hit to shooter
                RpcId(peerId, nameof(ConfirmHit), target.PeerId, damage, isHeadshot, hitPoint);
                
                // Broadcast hit to all (for effects)
                Rpc(nameof(SpawnHitEffect), hitPoint, hitNormal);
            }
            else
            {
                // Hit environment
                Rpc(nameof(SpawnImpactEffect), hitPoint, hitNormal);
            }
        }
    }

    public override void _Ready()
    {
        // Only interpolate remote players
        if (PeerId == NetworkManager.Instance?.LocalPeerId)
        {
            SetProcess(false);
            SetPhysicsProcess(false);

        _targetRotation = Rotation;

    public override void _PhysicsProcess(double delta)
    {
        if (NetworkManager.Instance?.LocalPeerId == PeerId) return;
        
        // Interpolate position
        GlobalPosition = GlobalPosition.Lerp(_targetPosition, (float)delta * InterpolationSpeed);
        Rotation = Rotation.Lerp(_targetRotation, (float)delta * InterpolationSpeed);
    }
    
    public void InterpolateState(PlayerState state)
    {
        // Add to buffer for jitter compensation
        _stateBuffer.Enqueue(state);
        
        if (_stateBuffer.Count > BUFFER_SIZE)
        {
            var targetState = _stateBuffer.Dequeue();
            _targetPosition = targetState.Position;
            _targetRotation = targetState.Rotation;
            _health = targetState.Health;
        }
    }

        }

public partial class SeasonPass : Node
{

        if (CurrentTier >= MAX_TIER) return;
        
        CurrentXP += amount;
        EmitSignal(SignalName.XPGained, amount, (int)source);
        
        while (CurrentTier < MAX_TIER && CurrentXP >= XPForNextTier)

            EmitSignal(SignalName.TierUnlocked, tier, false);

            GrantReward(reward.PremiumReward);
            _claimedRewards[tier * 2 + 1] = true;
            EmitSignal(SignalName.TierUnlocked, tier, true);

                GameManager.Instance?.AddCurrency(reward.Amount);
                break;
            case RewardType.Skin:
            case RewardType.Agent:
            case RewardType.Card:
            case RewardType.Spray:
            case RewardType.Buddy:
                GD.Print($"[SeasonPass] Unlocked: {reward.Name} ({reward.Rarity})");
                // Add to inventory
                break;

        if (HasPremium) return false;
        
        if (!GameManager.Instance.SpendCurrency(PREMIUM_PRICE))
            return false;
        
        HasPremium = true;
        
        // Grant retroactive rewards
        for (int i = 1; i <= CurrentTier; i++)
        {
            var premium = _rewards[i - 1].PremiumReward;
            if (premium != null && !_claimedRewards.GetValueOrDefault(i * 2 + 1))
            {
                GrantReward(premium);
                _claimedRewards[i * 2 + 1] = true;
            }
        }
        
        EmitSignal(SignalName.PremiumUnlocked);
        SaveProgress();
        return true;
    }
    
    public void PurchaseTierSkip(int tiers)
    {
        int maxSkip = MAX_TIER - CurrentTier;
        tiers = Math.Min(tiers, maxSkip);
        int price = tiers * 150; // 150 VP per tier
        
        if (!GameManager.Instance.SpendCurrency(price)) return;
        
        for (int i = 0; i < tiers; i++)
        {
            int xpNeeded = XPForNextTier - CurrentXP;
            AddXP(xpNeeded, XPSource.Purchase);
        }
    }

    {
        if (GameManager.Instance == null) return;
        
        GameManager.Instance.BattlePassTier = CurrentTier;
        GameManager.Instance.HasPremiumPass = HasPremium;
        GameManager.Instance.SavePlayerData();
    }

    {
        if (GameManager.Instance == null) return;
        
        CurrentTier = GameManager.Instance.BattlePassTier;
        HasPremium = GameManager.Instance.HasPremiumPass;
    }
