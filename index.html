    src/Core/GameManager.cs 
    using Godot;
using System;
using System.Collections.Generic;

public partial class GameManager : Node
{
    public static GameManager Instance { get; private set; }
    
    [Signal] public delegate void PlayerSpawnedEventHandler(PlayerController player);
    [Signal] public delegate void MatchStartedEventHandler(string matchId);
    [Signal] public delegate void MatchEndedEventHandler(string winner);
    [Signal] public delegate void CurrencyChangedEventHandler(int amount);
    
    // Player Data
    public PlayerData CurrentPlayer { get; set; } = new();
    public GameState State { get; private set; } = GameState.MainMenu;
    
    // Monetization
    public int PremiumCurrency { get; set; } = 1000;  // VP
    public HashSet<string> OwnedDLCs { get; set; } = new();
    public int BattlePassTier { get; set; } = 0;
    public bool HasPremiumPass { get; set; } = false;
    
    // Settings
    public float MouseSensitivity { get; set; } = 0.5f;
    public int TargetFPS { get; set; } = 144;
    public bool VSync { get; set; } = true;
    
    public override void _Ready()
    {
        Instance = this;
        ProcessMode = ProcessModeEnum.Always;
        
        // Check Vulkan support
        var renderingDevice = RenderingServer.GetRenderingDevice();
        if (renderingDevice == null)
        {
            PushError("Vulkan rendering device not available!");
            GetTree().Quit();
            return;
        }
        
        GD.Print($"GPU: {renderingDevice.GetDeviceName()}");
        GD.Print($"Driver: {renderingDevice.GetDriverVersion()}");
        
        LoadSettings();
        LoadPlayerData();
    }
    
    public void ChangeState(GameState newState)
    {
        var oldState = State;
        State = newState;
        GD.Print($"State changed: {oldState} -> {newState}");
    }
    
    public void AddCurrency(int amount)
    {
        PremiumCurrency += amount;
        EmitSignal(SignalName.CurrencyChanged, PremiumCurrency);
        SavePlayerData();
    }
    
    public bool SpendCurrency(int amount)
    {
        if (PremiumCurrency < amount) return false;
        PremiumCurrency -= amount;
        EmitSignal(SignalName.CurrencyChanged, PremiumCurrency);
        SavePlayerData();
        return true;
    }
    
    public void UnlockDLC(string dlcId)
    {
        OwnedDLCs.Add(dlcId);
        SavePlayerData();
    }
    
    public bool HasDLC(string dlcId) => OwnedDLCs.Contains(dlcId);
    
    private void LoadSettings()
    {
        // Load from user://settings.cfg
        var config = new ConfigFile();
        var err = config.Load("user://settings.cfg");
        if (err != Error.Ok) return;
        
        MouseSensitivity = (float)config.GetValue("input", "mouse_sensitivity", 0.5f);
        TargetFPS = (int)config.GetValue("video", "target_fps", 144);
        VSync = (bool)config.GetValue("video", "vsync", true);
        
        Engine.MaxFps = TargetFPS;
    }
    
    private void LoadPlayerData()
    {
        if (!FileAccess.FileExists("user://player.save")) return;
        
        using var file = FileAccess.Open("user://player.save", FileAccess.ModeFlags.Read);
        var json = file.GetAsText();
        var data = Json.ParseString(json).AsGodotDictionary();
        
        PremiumCurrency = (int)data.GetValueOrDefault("vp", 1000);
        BattlePassTier = (int)data.GetValueOrDefault("bp_tier", 0);
        HasPremiumPass = (bool)data.GetValueOrDefault("bp_premium", false);
        
        var dlcs = data.GetValueOrDefault("dlcs", new Godot.Collections.Array<string>());
        foreach (var dlc in dlcs)
            OwnedDLCs.Add(dlc.ToString());
    }
    
    public void SavePlayerData()
    {
        var data = new Godot.Collections.Dictionary
        {
            ["vp"] = PremiumCurrency,
            ["bp_tier"] = BattlePassTier,
            ["bp_premium"] = HasPremiumPass,
            ["dlcs"] = new Godot.Collections.Array<string>(OwnedDLCs)
        };
        
        using var file = FileAccess.Open("user://player.save", FileAccess.ModeFlags.Write);
        file.StoreString(Json.Stringify(data));
    }
    
    public override void _Notification(int what)
    {
        if (what == NotificationWMCloseRequest)
        {
            SavePlayerData();
            GetTree().Quit();
        }
    }
}

    public PlayerStats Stats { get; set; } = new();

public class PlayerStats
{
    public int Kills { get; set; }
    public int Deaths { get; set; }
    public int Assists { get; set; }
    public int Wins { get; set; }
    public int Losses { get; set; }
    public int Headshots { get; set; }
    public int DamageDealt { get; set; }
    
    public float KDRatio => Deaths == 0 ? Kills : (float)Kills / Deaths;
    public float WinRate => (Wins + Losses) == 0 ? 0 : (float)Wins / (Wins + Losses);
    public float HeadshotRate => Kills == 0 ? 0 : (float)Headshots / Kills;
}

    3.2 PlayerController.cs

    src/Player/PlayerController.cs 
    using Godot;
using System;

[GlobalClass]
public partial class PlayerController : CharacterBody3D
{
    [Export] public float WalkSpeed = 5.0f;
    [Export] public float SprintSpeed = 8.0f;
    [Export] public float CrouchSpeed = 2.5f;
    [Export] public float JumpVelocity = 4.5f;
    [Export] public float MouseSensitivity = 0.005f;
    
    [ExportGroup("Camera")]
    [Export] public Camera3D Camera;
    [Export] public Node3D CameraPivot;
    [Export] public ShapeCast3D CrouchCheck;
    
    [ExportGroup("Weapons")]
    [Export] public Node3D WeaponHolder;
    [Export] public RayCast3D AimRay;
    
    // State
    public bool IsSprinting { get; private set; }
    public bool IsCrouching { get; private set; }
    public bool IsAiming { get; private set; }
    public bool IsGrounded => IsOnFloor();
    
    private float _gravity = ProjectSettings.GetSetting("physics/3d/default_gravity").AsSingle();
    private float _currentSpeed;
    private float _cameraVerticalRotation = 0f;
    private Vector3 _velocity;
    
    // Components
    private WeaponManager _weaponManager;
    private HealthSystem _healthSystem;
    private AnimationPlayer _animPlayer;
    
    public override void _Ready()
    {
        _currentSpeed = WalkSpeed;
        _weaponManager = GetNode<WeaponManager>("WeaponManager");
        _healthSystem = GetNode<HealthSystem>("HealthSystem");
        _animPlayer = GetNode<AnimationPlayer>("AnimationPlayer");
        
        // Lock mouse
        Input.MouseMode = Input.MouseModeEnum.Captured;
        
        // Load sensitivity from settings
        MouseSensitivity = GameManager.Instance.MouseSensitivity;
    }
    
    public override void _Input(InputEvent @event)
    {
        if (@event is InputEventMouseMotion mouseMotion && Input.MouseMode == Input.MouseModeEnum.Captured)
        {
            HandleMouseLook(mouseMotion);
        }
        
        if (@event.IsActionPressed("ui_cancel"))
        {
            Input.MouseMode = Input.MouseMode == Input.MouseModeEnum.Captured 
                ? Input.MouseModeEnum.Visible 
                : Input.MouseModeEnum.Captured;
        }
    }
    
    private void HandleMouseLook(InputEventMouseMotion motion)
    {
        // Horizontal rotation (player body)
        RotateY(-motion.Relative.X * MouseSensitivity);
        
        // Vertical rotation (camera)
        _cameraVerticalRotation -= motion.Relative.Y * MouseSensitivity;
        _cameraVerticalRotation = Mathf.Clamp(_cameraVerticalRotation, -Mathf.Pi / 2, Mathf.Pi / 2);
        CameraPivot.Rotation = new Vector3(_cameraVerticalRotation, 0, 0);
    }
    
    public override void _PhysicsProcess(double delta)
    {
        HandleMovement(delta);
        HandleActions();
        ApplyGravity(delta);
        
        Velocity = _velocity;
        MoveAndSlide();
        
        // Sync weapon with camera
        _weaponManager?.UpdateWeaponPosition(IsAiming, (float)delta);
    }
    
    private void HandleMovement(double delta)
    {
        Vector3 direction = Vector3.Zero;
        
        // Get input
        if (Input.IsActionPressed("move_forward")) direction -= Transform.Basis.Z;
        if (Input.IsActionPressed("move_backward")) direction += Transform.Basis.Z;
        if (Input.IsActionPressed("move_left")) direction -= Transform.Basis.X;
        if (Input.IsActionPressed("move_right")) direction += Transform.Basis.X;
        
        direction = direction.Normalized();
        
        // Sprint
        IsSprinting = Input.IsActionPressed("sprint") && !IsCrouching && direction != Vector3.Zero;
        _currentSpeed = IsSprinting ? SprintSpeed : (IsCrouching ? CrouchSpeed : WalkSpeed);
        
        // Crouch
        if (Input.IsActionJustPressed("crouch"))
        {
            ToggleCrouch();
        }
        
        // Apply movement
        if (direction != Vector3.Zero)
        {
            _velocity.X = direction.X * _currentSpeed;
            _velocity.Z = direction.Z * _currentSpeed;
            
            // Weapon bob
            _weaponManager?.ApplyWeaponBob(_currentSpeed, IsSprinting);
        }
        else
        {
            _velocity.X = Mathf.MoveToward(_velocity.X, 0, _currentSpeed);
            _velocity.Z = Mathf.MoveToward(_velocity.Z, 0, _currentSpeed);
        }
        
        // Jump
        if (Input.IsActionJustPressed("jump") && IsOnFloor() && !IsCrouching)
        {
            _velocity.Y = JumpVelocity;
            _animPlayer?.Play("jump");
        }
    }
    
    private void HandleActions()
    {
        // Aim
        IsAiming = Input.IsActionPressed("aim");
        
        // Shoot
        if (Input.IsActionPressed("shoot"))
        {
            _weaponManager?.TryShoot(IsAiming);
        }
        
        // Reload
        if (Input.IsActionJustPressed("reload"))
        {
            _weaponManager?.Reload();
        }
        
        // Weapon switch
        if (Input.IsActionJustPressed("weapon_1")) _weaponManager?.SwitchWeapon(0);
        if (Input.IsActionJustPressed("weapon_2")) _weaponManager?.SwitchWeapon(1);
        if (Input.IsActionJustPressed("weapon_3")) _weaponManager?.SwitchWeapon(2);
    }
    
    private void ApplyGravity(double delta)
    {
        if (!IsOnFloor())
        {
            _velocity.Y -= _gravity * (float)delta;
        }
        else if (_velocity.Y < 0)
        {
            _velocity.Y = 0;
        }
    }
    
    private void ToggleCrouch()
    {
        IsCrouching = !IsCrouching;
        
        // Check if can stand up
        if (!IsCrouching && CrouchCheck.IsColliding())
        {
            IsCrouching = true;
            return;
        }
        
        // Adjust collision shape
        var collision = GetNode<CollisionShape3D>("CollisionShape3D");
        var shape = collision.Shape as CapsuleShape3D;
        
        if (IsCrouching)
        {
            shape.Height = 1.0f;
            collision.Position = new Vector3(0, 0.5f, 0);
        }
        else
        {
            shape.Height = 1.8f;
            collision.Position = new Vector3(0, 0.9f, 0);
        }
        
        _animPlayer?.Play(IsCrouching ? "crouch" : "uncrouch");
    }
    
    public void TakeDamage(int damage, Vector3 hitPosition, Node3D attacker)
    {
        _healthSystem?.TakeDamage(damage);
        
        // Hit marker effect
        Camera.Shake(0.2f, 0.1f);
        
        // Blood effect
        SpawnBloodEffect(hitPosition);
        
        if (_healthSystem?.CurrentHealth <= 0)
        {
            Die(attacker);
        }
    }
    
    private void Die(Node3D killer)
    {
        GameManager.Instance.CurrentPlayer.Stats.Deaths++;
        
        // Death cam
        Camera.ClearCurrent();
        
        // Respawn or spectate
        EmitSignal(SignalName.Died, killer);
    }
    
    private void SpawnBloodEffect(Vector3 position)
    {
        var blood = GD.Load<PackedScene>("res://effects/blood_splash.tscn");
        var instance = blood.Instantiate<GpuParticles3D>();
        instance.GlobalPosition = position;
        GetTree().CurrentScene.AddChild(instance);
    }
    
    [Signal] public delegate void DiedEventHandler(Node3D killer);
}

    using Godot;
using System;

[GlobalClass]
public abstract partial class BaseWeapon : Node3D
{
    [ExportGroup("Stats")]
    [Export] public string WeaponName = "Weapon";
    [Export] public int Damage = 25;
    [Export] public int HeadshotDamage = 75;
    [Export] public int FireRate = 600;  // RPM
    [Export] public int MagazineSize = 30;
    [Export] public int ReserveAmmo = 90;
    [Export] public float ReloadTime = 2.0f;
    [Export] public float Range = 100f;
    [Export] public WeaponType Type = WeaponType.Rifle;
    
    [ExportGroup("Recoil")]
    [Export] public float RecoilVertical = 0.5f;
    [Export] public float RecoilHorizontal = 0.2f;
    [Export] public float RecoilRecovery = 5f;
    
    [ExportGroup("Spread")]
    [Export] public float BaseSpread = 0.01f;
    [Export] public float MoveSpread = 0.05f;
    [Export] public float JumpSpread = 0.1f;
    
    [ExportGroup("Audio")]
    [Export] public AudioStream ShootSound;
    [Export] public AudioStream ReloadSound;
    [Export] public AudioStream EmptySound;
    
    // State
    public int CurrentAmmo { get; protected set; }
    public int CurrentReserve { get; protected set; }
    public bool IsReloading { get; protected set; }
    public bool CanShoot => !IsReloading && CurrentAmmo > 0 && _fireTimer <= 0;
    
    protected float _fireInterval;
    protected float _fireTimer = 0f;
    protected float _recoilAccumulation = 0f;
    protected RandomNumberGenerator _rng = new();
    
    // Components
    protected AudioStreamPlayer3D _audioPlayer;
    protected AnimationPlayer _animPlayer;
    protected GpuParticles3D _muzzleFlash;
    protected RayCast3D _fireRay;
    
    public override void _Ready()
    {
        _fireInterval = 60f / FireRate;
        CurrentAmmo = MagazineSize;
        CurrentReserve = ReserveAmmo;
        
        _audioPlayer = GetNode<AudioStreamPlayer3D>("AudioStreamPlayer3D");
        _animPlayer = GetNode<AnimationPlayer>("AnimationPlayer");
        _muzzleFlash = GetNode<GpuParticles3D>("MuzzleFlash");
        _fireRay = GetNode<RayCast3D>("FireRay");
    }
    
    public override void _Process(double delta)
    {
        if (_fireTimer > 0)
            _fireTimer -= (float)delta;
        
        // Recoil recovery
        _recoilAccumulation = Mathf.MoveToward(_recoilAccumulation, 0, RecoilRecovery * (float)delta);
    }
    
    public virtual bool TryShoot(bool isAiming)
    {
        if (!CanShoot) 
        {
            if (CurrentAmmo == 0)
                PlayEmptySound();
            return false;
        }
        
        _fireTimer = _fireInterval;
        
        // Consume ammo
        CurrentAmmo--;
        
        // Effects
        PlayShootEffects();
        
        // Raycast
        PerformRaycast(isAiming);
        
        // Recoil
        ApplyRecoil();
        
        return true;
    }
    
    protected virtual void PerformRaycast(bool isAiming)
    {
        var spread = CalculateSpread(isAiming);
        _fireRay.Rotation = new Vector3(
            _rng.Randf() * spread - spread / 2,
            _rng.Randf() * spread - spread / 2,
            0
        );
        
        _fireRay.ForceRaycastUpdate();
        
        if (_fireRay.IsColliding())
        {
            var hit = _fireRay.GetCollider() as Node3D;
            var hitPoint = _fireRay.GetCollisionPoint();
            var normal = _fireRay.GetCollisionNormal();
            
            // Spawn impact effect
            SpawnImpactEffect(hitPoint, normal);
            
            // Deal damage
            if (hit is PlayerController player)
            {
                bool isHeadshot = IsHeadshot(hitPoint, player);
                int damage = isHeadshot ? HeadshotDamage : Damage;
                
                player.TakeDamage(damage, hitPoint, GetParent().GetParent() as Node3D);
                
                // Hit marker
                ShowHitMarker(isHeadshot);
            }
        }
    }
    
    protected float CalculateSpread(bool isAiming)
    {
        float spread = BaseSpread;
        
        // Add movement spread
        var player = GetParent().GetParent() as PlayerController;
        if (player != null)
        {
            if (!player.IsGrounded) spread += JumpSpread;
            if (player.Velocity.Length() > 0.1f) spread += MoveSpread;
        }
        
        // Aiming reduces spread
        if (isAiming) spread *= 0.5f;
        
        // Add accumulated recoil
        spread += _recoilAccumulation * 0.1f;
        
        return spread;
    }
    
    protected void ApplyRecoil()
    {

        if (camera != null)
        {
            camera.Rotation -= new Vector3(RecoilVertical, 
                _rng.Randf() * RecoilHorizontal * 2 - RecoilHorizontal, 0);
        }
    }
    
    public virtual void Reload()
    {
        if (IsReloading || CurrentAmmo >= MagazineSize || CurrentReserve <= 0)
            return;
        
        IsReloading = true;
        _animPlayer?.Play("reload");
        _audioPlayer?.Stream = ReloadSound;
        _audioPlayer?.Play();
        
        GetTree().CreateTimer(ReloadTime).Timeout += FinishReload;
    }
    
    protected virtual void FinishReload()
    {
        int ammoNeeded = MagazineSize - CurrentAmmo;
        int ammoToLoad = Mathf.Min(ammoNeeded, CurrentReserve);
        
        CurrentAmmo += ammoToLoad;
        CurrentReserve -= ammoToLoad;
        
        IsReloading = false;
    }
    
    protected void PlayShootEffects()
    {
        _animPlayer?.Play("shoot");
        _audioPlayer?.Stream = ShootSound;
        _audioPlayer?.Play();
        _muzzleFlash?.Restart();
        _muzzleFlash?.Emitting = true;
    }

        _audioPlayer?.Stream = EmptySound;
        _audioPlayer?.Play();
    }

        if (IsServer)
        {
            // Broadcast to all clients
            Rpc(nameof(SyncPlayerList), _players);
            
            // Spawn player in world
            SpawnPlayer(peerId, info);
        }

    {
        var spawnPoints = GetTree().GetNodesInGroup("SpawnPoints");
        var spawnPoint = spawnPoints[_players.Count % spawnPoints.Count] as Node3D;
        
        var playerScene = GD.Load<PackedScene>("res://player/multiplayer_player.tscn");
        var player = playerScene.Instantiate<MultiplayerPlayer>();
        player.Name = peerId.ToString();
        player.PeerId = peerId;
        player.PlayerInfo = info;
        player.GlobalPosition = spawnPoint?.GlobalPosition ?? Vector3.Zero;
        
        GetNode("/root/GameWorld/Players").AddChild(player, true); // Networked spawn
    }
    
    // Server-authoritative shooting
    [Rpc(MultiplayerApi.RpcMode.AnyPeer, CallLocal = false)]
    public void RequestShoot(int peerId, Vector3 origin, Vector3 direction)
    {
        if (!IsServer) return;
        
        // Validate on server
        var shooter = GetNodeOrNull<MultiplayerPlayer>($"/root/GameWorld/Players/{peerId}");
        if (shooter == null) return;
        
        // Perform raycast
        var spaceState = GetWorld3D().DirectSpaceState;
        var query = PhysicsRayQueryParameters3D.Create(origin, origin + direction * 1000);
        query.CollisionMask = 0b101; // Players and environment
        
        var result = spaceState.IntersectRay(query);
        
        if (result.Count > 0)
        {
            var hit = result["collider"].AsGodotObject() as Node3D;
            var hitPoint = result["position"].AsVector3();
            
            if (hit is MultiplayerPlayer target && target.PeerId != peerId)
            {
                // Apply damage
                int damage = CalculateDamage(hitPoint, target);
                target.Rpc(nameof(target.TakeDamage), damage, peerId);
                
                // Confirm hit to shooter
                RpcId(peerId, nameof(ConfirmHit), target.PeerId, damage, true);
            }
        }
    }
    
    [Rpc(MultiplayerApi.RpcMode.Authority)]
    private void ConfirmHit(int targetId, int damage, bool isKill)
    {
        // Show hit marker on client
        GD.Print($"Hit confirmed: {damage} damage");
    }
    
    private int CalculateDamage(Vector3 hitPoint, MultiplayerPlayer target)
    {
        // Headshot check
        var headPos = target.GlobalPosition + new Vector3(0, 1.6f, 0);
        if (hitPoint.DistanceTo(headPos) < 0.15f)
            return 75; // Headshot
        
        return 25; // Body shot
    }
}

    using Godot;
using System;
using System.Collections.Generic;

public partial class SeasonPass : Node
{
    [Signal] public delegate void TierUnlockedEventHandler(int tier);
    [Signal] public delegate void PremiumUnlockedEventHandler();

    public int CurrentSeason { get; set; } = 1;
    public string SeasonName { get; set; } = "First Blood";
    public DateTime SeasonEnd { get; set; } = new DateTime(2026, 6, 28);
    
    public int CurrentTier { get; private set; } = 0;
    public int CurrentXP { get; private set; } = 0;
    public bool HasPremium { get; set; } = false;
    public const int MAX_TIER = 100;
    
    private List<SeasonReward> _rewards = new();
    
    public override void _Ready()
    {
        InitializeRewards();
        LoadProgress();
    }
    
    private void InitializeRewards()
    {
        for (int tier = 1; tier <= MAX_TIER; tier++)
        {
            _rewards.Add(new SeasonReward
            {
                Tier = tier,
                FreeReward = GenerateFreeReward(tier),
                PremiumReward = GeneratePremiumReward(tier),
                XpRequired = CalculateXP(tier)
            });
        }
    }
    
    private Reward GenerateFreeReward(int tier)
    {
        return tier switch
        {
            1 => new Reward(RewardType.Currency, 100, "100 VP"),
            10 => new Reward(RewardType.Skin, 1, "Classic S1"),
            25 => new Reward(RewardType.Agent, 1, "Agent Token"),
            50 => new Reward(RewardType.Skin, 1, "AK-47 First Blood", Rarity.Epic),
            100 => new Reward(RewardType.Skin, 1, "Karambit Bloodied", Rarity.Legendary),
            _ => new Reward(RewardType.Currency, tier % 5 == 0 ? 50 : 25, $"{(tier % 5 == 0 ? 50 : 25)} VP")
        };
    }
    
    private Reward GeneratePremiumReward(int tier)
    {
        return tier switch
        {
            1 => new Reward(RewardType.Skin, 1, "Premium S1", Rarity.Rare),
            50 => new Reward(RewardType.Skin, 1, "AWP Dragon Lite", Rarity.Legendary),
            100 => new Reward(RewardType.Skin, 1, "Karambit Fade", Rarity.Mythic),
            _ => new Reward(RewardType.Currency, tier * 5, $"{tier * 5} VP")
        };
    }
    
    private int CalculateXP(int tier) => (int)(1000 * Math.Pow(tier, 1.2));
    
    public void AddXP(int amount, XPSource source)
    {
        if (CurrentTier >= MAX_TIER) return;
        
        CurrentXP += amount;
        
        while (CurrentTier < MAX_TIER && CurrentXP >= XpForNextTier)
        {
            CurrentXP -= XpForNextTier;
            UnlockTier(++CurrentTier);
        }
        
        SaveProgress();
    }
    
    private void UnlockTier(int tier)
    {
        var reward = _rewards[tier - 1];
        GrantReward(reward.FreeReward);
        
        if (HasPremium && reward.PremiumReward != null)
            GrantReward(reward.PremiumReward);
        
        EmitSignal(SignalName.TierUnlocked, tier);
        
        if (tier == MAX_TIER)
            EmitSignal(SignalName.SeasonCompleted);
    }
    
    private void GrantReward(Reward reward)
    {
        switch (reward.Type)
        {
            case RewardType.Currency:
                GameManager.Instance.AddCurrency(reward.Amount);
                break;
            case RewardType.Skin:
            case RewardType.Agent:
                GD.Print($"Unlocked: {reward.Name}");
                break;
        }
    }
    
    public bool PurchasePremium()
    {
        const int PRICE = 1000;
        if (!GameManager.Instance.SpendCurrency(PRICE)) return false;
        
        HasPremium = true;
        
        // Retroactive rewards
        for (int i = 1; i <= CurrentTier; i++)
        {
            var premium = _rewards[i - 1].PremiumReward;
            if (premium != null) GrantReward(premium);
        }
        
        EmitSignal(SignalName.PremiumUnlocked);
        SaveProgress();
        return true;
    }
    
    public void PurchaseTierSkip(int tiers)
    {
        int maxSkip = MAX_TIER - CurrentTier;
        tiers = Math.Min(tiers, maxSkip);
        int price = tiers * 150;
        
        if (!GameManager.Instance.SpendCurrency(price)) return;
        
        for (int i = 0; i < tiers; i++)
            AddXP(XpForNextTier - CurrentXP, XPSource.Purchase);
    }
    
    private void SaveProgress()
    {
        GameManager.Instance.BattlePassTier = CurrentTier;
        GameManager.Instance.HasPremiumPass = HasPremium;
        GameManager.Instance.SavePlayerData();
    }
    
    private void LoadProgress()
    {
        CurrentTier = GameManager.Instance.BattlePassTier;
        HasPremium = GameManager.Instance.HasPremiumPass;
    }
    
    public int XpForNextTier => CalculateXP(CurrentTier + 1);
    public TimeSpan TimeRemaining => SeasonEnd - DateTime.UtcNow;
}
